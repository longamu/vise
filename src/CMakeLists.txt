MACRO(MY_EVAL_CONDITION name)
   IF(${ARGN})
     SET(${name} 1)
   ELSE(${ARGN})
     SET(${name} 0)
   ENDIF(${ARGN})
ENDMACRO(MY_EVAL_CONDITION)

MACRO(MY_OPTION name description value)
   IF(${value})
     OPTION(${name} description ON)
   ELSE(${value})
     OPTION(${name} description 0FF)
   ENDIF(${value})
ENDMACRO(MY_OPTION)
